\chapter{Implementation and Testing}
This chapter presents core algorithms, external API integrations, and testing strategies for the final iteration. The implemented modules include: User Authentication, Pet Breed Identification, AI Chatbot, Veterinary Appointments, Pet Toy Marketplace, Community Hub (Forum, Lost \& Found, Adoption, Events), Pet Caregiver Services with GPS Tracking, Admin Portal, Payment Integration (Cryptocurrency + Wallet), Chat \& Messaging, and Backend Infrastructure.

\section{Algorithm Design}

\subsection{Algorithm 1: RAG-based Chatbot Query Processing}

\begin{algorithmbox}{RAG-based Chatbot Query Processing}
\KwIn{$userQuery$ (string), $petProfile$ (optional dictionary)}
\KwOut{$response$ (string), $sources$ (list)}
\aln{1}{$query \leftarrow userQuery$}
\aln{2}{\textbf{if} $petProfile$ exists \textbf{then}}
\aln{3}{\I $query \leftarrow$ "Pet: " $+ petProfile.type +$ ", " $+ petProfile.breed +$ ", " $+ petProfile.age +$ "\textbackslash n" $+ userQuery$}
\aln{4}{\textbf{end if}}
\aln{5}{$retriever \leftarrow$ VectorDB.AsRetriever(similarity, $k=2$)}
\aln{6}{$docs \leftarrow retriever$.Invoke($query$)}
\aln{7}{$context \leftarrow$ FormatDocs($docs$)}
\aln{8}{$prompt \leftarrow$ BuildPrompt("PawPal veterinary AI", $context$, $query$)}
\aln{9}{$response \leftarrow$ GroqAPI.ChatCompletion("llama-3.3-70b-versatile", $prompt$, $maxTokens{=}384$, $temp{=}0.3$)}
\aln{10}{$sources \leftarrow [\,]$}
\aln{11}{\textbf{for each} $doc$ \textbf{in} $docs[0:2]$ \textbf{do}}
\aln{12}{\I $sources$.append($\{content: doc.content[0:150],\ meta: doc.metadata\}$)}
\aln{13}{\textbf{end for}}
\aln{14}{\textbf{return} $(response,\ sources)$}
\end{algorithmbox}

\subsection{Algorithm 2: CNN-based Breed Identification}

\begin{algorithmbox}{CNN-based Breed Identification}
\KwIn{$imageData$ (binary image file)}
\KwOut{$breedPredictions$ (list), $careRecommendations$ (dict)}
\aln{1}{\textbf{if} $\neg$ValidateImage($imageData$) \textbf{then}}
\aln{2}{\I \textbf{return} Error("Invalid format or size $> 10$MB")}
\aln{3}{\textbf{end if}}
\aln{4}{$image \leftarrow$ DecodeImage($imageData$)}
\aln{5}{$img \leftarrow$ ResizeImage($image$, $384 \times 384$)}
\aln{6}{$mean \leftarrow [0.485,\ 0.456,\ 0.406]$;\enspace $std \leftarrow [0.229,\ 0.224,\ 0.225]$}
\aln{7}{\textbf{for each} channel $ch \in \{R, G, B\}$ \textbf{do}}
\aln{8}{\I $img[ch] \leftarrow (img[ch] / 255.0 - mean[ch]) / std[ch]$}
\aln{9}{\textbf{end for}}
\aln{10}{$batch \leftarrow$ ExpandDims($img$, axis $= 0$)}
\aln{11}{$species \leftarrow$ ClassifySpecies($image$)}
\aln{12}{\textbf{if} $species =$ "dog" \textbf{then}}
\aln{13}{\I $model \leftarrow$ LoadModel("convnextv2\_dog\_120breeds")}
\aln{14}{\textbf{else}}
\aln{15}{\I $model \leftarrow$ LoadModel("convnextv2\_cat\_37breeds")}
\aln{16}{\textbf{end if}}
\aln{17}{$preds \leftarrow model$.predict($batch$)}
\aln{18}{$topIdx \leftarrow$ ArgTopK($preds$, $k = 3$)}
\aln{19}{$breedPredictions \leftarrow [\,]$}
\aln{20}{\textbf{for each} $i \in topIdx$ \textbf{do}}
\aln{21}{\I $breedPredictions$.append($\{breed: labels[i],\ conf: preds[i] \times 100\}$)}
\aln{22}{\textbf{end for}}
\aln{23}{\textbf{if} $breedPredictions[0].conf \geq 85.0$ \textbf{then}}
\aln{24}{\I $careRecommendations \leftarrow$ DB.GetBreedInfo($breedPredictions[0].breed$)}
\aln{25}{\textbf{else}}
\aln{26}{\I $careRecommendations \leftarrow$ GetGeneralCareInfo($species$)}
\aln{27}{\textbf{end if}}
\aln{28}{DB.LogBreedIdentification($species$, $breedPredictions$)}
\aln{29}{\textbf{return} $(breedPredictions,\ careRecommendations)$}
\end{algorithmbox}

\subsection{Algorithm 3: Veterinary Appointment Booking with GPS Discovery}

\begin{algorithmbox}{Veterinary Appointment Booking with GPS Discovery}
\KwIn{$petID$, $userLocation$ (lat, lng), $preferredDate$}
\KwOut{$appointment$, $nearbyClinics$ (list sorted by distance)}
\aln{1}{$pet \leftarrow$ DB.GetPetProfile($petID$)}
\aln{2}{$owner \leftarrow$ DB.GetUser($pet.ownerID$)}
\aln{3}{$osmResults \leftarrow$ OSM.Search($userLocation$, radius $= 10$km, type $=$ "vet")}
\aln{4}{$nearbyClinics \leftarrow [\,]$}
\aln{5}{\textbf{for each} $clinic$ \textbf{in} $osmResults$ \textbf{do}}
\aln{6}{\I $details \leftarrow$ DB.GetClinicByOSMID($clinic.osmID$)}
\aln{7}{\I \textbf{if} $details$ exists \textbf{then}}
\aln{8}{\II $dist \leftarrow$ Haversine($userLocation$, $clinic.location$)}
\aln{9}{\II $nearbyClinics$.append($\{id,\ name,\ address,\ rating,\ dist\}$)}
\aln{10}{\I \textbf{end if}}
\aln{11}{\textbf{end for}}
\aln{12}{$nearbyClinics$.SortBy($dist$)}
\aln{13}{$slots \leftarrow$ DB.GetAvailableSlots($selectedClinic.id$, $preferredDate$)}
\aln{14}{$appt \leftarrow$ DB.CreateAppointment($petID$, $owner.id$, $selectedSlot$, "Requested")}
\aln{15}{NotificationSvc.NotifyClinic($selectedClinic.id$, $appt.id$)}
\aln{16}{NotificationSvc.ScheduleReminder($appt.id$, $appt.dateTime - 24\text{h}$)}
\aln{17}{NotificationSvc.ScheduleReminder($appt.id$, $appt.dateTime - 2\text{h}$)}
\aln{18}{\textbf{return} $(appt,\ nearbyClinics)$}
\end{algorithmbox}

\begin{algorithmbox}{Haversine Distance Calculation}
\KwIn{$lat_1, lng_1, lat_2, lng_2$ (decimal degrees)}
\KwOut{$distance$ (kilometres)}
\aln{1}{$R \leftarrow 6371$}
\aln{2}{$\Delta lat \leftarrow \mathrm{toRad}(lat_2 - lat_1)$}
\aln{3}{$\Delta lng \leftarrow \mathrm{toRad}(lng_2 - lng_1)$}
\aln{4}{$a \leftarrow \sin^2(\Delta lat/2) + \cos(lat_1)\cdot\cos(lat_2)\cdot\sin^2(\Delta lng/2)$}
\aln{5}{$c \leftarrow 2\cdot\mathrm{atan2}(\sqrt{a},\,\sqrt{1-a})$}
\aln{6}{\textbf{return} $R \times c$}
\end{algorithmbox}

\subsection{Algorithm 4: Caregiver Booking and Live GPS Tracking}

\begin{algorithmbox}{Caregiver Booking and Live GPS Tracking}
\KwIn{$ownerID$, $caregiverID$, $serviceType$, $scheduledDate$}
\KwOut{$booking$, $routeHistory$}
\aln{1}{$cg \leftarrow$ DB.GetCaregiverProfile($caregiverID$)}
\aln{2}{\textbf{if} $\neg\,cg$.isAvailable($scheduledDate$) \textbf{then}}
\aln{3}{\I \textbf{return} Error("Caregiver not available")}
\aln{4}{\textbf{end if}}
\aln{5}{$booking \leftarrow$ DB.CreateBooking($ownerID$, $caregiverID$, $serviceType$, $scheduledDate$, "Requested")}
\aln{6}{NotificationSvc.Notify($caregiverID$, "New booking request")}
\aln{7}{\textbf{if} $booking.status =$ "Confirmed" \textbf{then} \quad\textit{// caregiver accepted}}
\aln{8}{\I $session \leftarrow$ DB.CreateGPSSession($booking.id$)}
\aln{9}{\I \textbf{while} $booking.status \neq$ "Completed" \textbf{do} \quad\textit{// update every 2 min}}
\aln{10}{\II $gps \leftarrow$ AwaitLocationUpdate(interval $= 120$s)}
\aln{11}{\II DB.SaveWaypoint($session.id$, $gps.lat$, $gps.lng$, Now())}
\aln{12}{\II BroadcastToOwner($ownerID$, $gps$)}
\aln{13}{\I \textbf{end while}}
\aln{14}{\I DB.CloseGPSSession($session.id$)}
\aln{15}{\I $routeHistory \leftarrow$ DB.GetFullRoute($session.id$)}
\aln{16}{\textbf{end if}}
\aln{17}{\textbf{return} $(booking,\ routeHistory)$}
\end{algorithmbox}

\subsection{Algorithm 5: Cryptocurrency Payment Processing}

\begin{algorithmbox}{Cryptocurrency Payment Processing}
\KwIn{$payerID$, $amount$, $currency \in \{\text{BTC, ETH, USDT}\}$, $refType$, $refID$}
\KwOut{$transaction$, $invoice$}
\aln{1}{$addr \leftarrow$ Blockchain.GenerateDepositAddress($currency$)}
\aln{2}{$qr \leftarrow$ GenerateQRCode($addr$, $amount$)}
\aln{3}{$txn \leftarrow$ DB.CreateTransaction($payerID$, $amount$, $currency$, $addr$, "Pending")}
\aln{4}{DisplayQR($qr$, $addr$, $amount$)}
\aln{5}{$confirmed \leftarrow \mathbf{false}$;\enspace $attempts \leftarrow 0$}
\aln{6}{\textbf{while} $\neg\,confirmed\ \wedge\ attempts < 30$ \textbf{do} \quad\textit{// poll every 60s, max 30 min}}
\aln{7}{\I $txHash \leftarrow$ Blockchain.CheckIncoming($addr$, $amount$)}
\aln{8}{\I \textbf{if} $txHash$ exists \textbf{then}}
\aln{9}{\II \textbf{if} Blockchain.GetConfirmations($txHash$) $\geq 1$ \textbf{then}}
\aln{10}{\III $confirmed \leftarrow \mathbf{true}$}
\aln{11}{\II \textbf{end if}}
\aln{12}{\I \textbf{end if}}
\aln{13}{\I Sleep($60$s);\enspace $attempts \leftarrow attempts + 1$}
\aln{14}{\textbf{end while}}
\aln{15}{\textbf{if} $confirmed$ \textbf{then}}
\aln{16}{\I DB.UpdateTransaction($txn.id$, "Completed", $txHash$)}
\aln{17}{\I UpdateStatus($refType$, $refID$, "Paid")}
\aln{18}{\I $invoice \leftarrow$ GenerateInvoice($txn$)}
\aln{19}{\I NotificationSvc.Notify($payerID$, "Payment confirmed")}
\aln{20}{\I \textbf{return} $(txn,\ invoice)$}
\aln{21}{\textbf{else}}
\aln{22}{\I DB.UpdateTransaction($txn.id$, "Failed")}
\aln{23}{\I \textbf{return} Error("Payment timeout --- no confirmation received")}
\aln{24}{\textbf{end if}}
\end{algorithmbox}
